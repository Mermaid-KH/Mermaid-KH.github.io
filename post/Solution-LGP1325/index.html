[Solution] LGP1325

发布于 2021-07-07 | 分类于 Solution | 5分钟 | 684字数

有平面上 nnn 个点 Pi(xi,yi)P_i(x_i, y_i)Pi​(xi​,yi​)，要求在 xxx 轴上取一些点 DiD_iDi​，使 ∀Pi,∃j,∣PiDj∣⩽d\forall P_i, \exists j, |P_iD_j| \leqslant d∀Pi​,∃j,∣Pi​Dj​∣⩽d。求所需点 DiD_iDi​ 的最小数目。

1⩽n⩽1031 \leqslant n \leqslant 10^31⩽n⩽103。

考虑根据 PiP_iPi​ 坐标可确定 ∣PiDj∣⩽d|P_iD_j| \leqslant d∣Pi​Dj​∣⩽d 的 DjD_jDj​ 的范围，设这个范围为 [li,ri][l_i, r_i][li​,ri​]。

将 PiP_iPi​ 根据 rir_iri​ 排序，然后每次找到第一个未被标记的点 PiP_iPi​，新建 Dj(ri,0)D_j(r_i, 0)Dj​(ri​,0)，将所有 ∣PkDj∣⩽d|P_kD_j| \leqslant d∣Pk​Dj​∣⩽d 的 PkP_kPk​ 标记。

正确性较显然。时间复杂度 O(n2)\mathcal O(n^2)O(n2)，空间复杂度 O(n)\mathcal O(n)O(n)。

# include <ciso646>
# include <cmath>
# include <cstdio>

namespace Main {
    typedef long long signed int lld;
    typedef short unsigned int hu;
    typedef unsigned int uint;
    typedef long long unsigned int llu;
    typedef double lf;
    static inline const llu abs(const lld x) { return x < 0 ? -x : x; }
    template <typename T> static inline const void swap(T& a, T& b) { const T c(a); a = b, b = c; }
    template <typename T> static inline const bool std_comp(const T& a, const T& b) { return a < b; }
    template <typename T>
    static inline const void sort
    (T* const b, T* const e, const bool (*comp)(const T&, const T&) = std_comp) {
        T* const a(b - 1); const uint l(e - b);
        for (register uint i(1); i <= l; ++i)
        { uint t(i); while (t > 1 and comp(a[t >> 1], a[t])) swap(a[t], a[t >> 1]), t >>= 1; }
        for (register uint i(l); i >= 1; --i) {
            swap(a[1], a[i]); uint t(1);
            while
            (t << 1 < i && comp(a[t], a[t << 1]) or (t << 1 | 1) < i && comp(a[t], a[t << 1 | 1]))
                (t << 1 | 1) < i && comp(a[t << 1], a[t << 1 | 1]) ?
                (swap(a[t], a[t << 1 | 1]), t = t << 1 | 1) : (swap(a[t], a[t << 1]), t <<= 1);
        }
    }
    enum { N = (const hu)1e3 };
    struct point {
        lld x, y;
        bool vis;
        point() : vis(false) {}
        point(const lld x, const lld y) : x(x), y(y) {}
        inline point& read() { scanf("%lld%lld", &x, &y); return *this; }
        inline const lf l() const; inline const lf r() const;
        friend inline const bool operator< (const point& a, const point& b) { return a.r() < b.r(); }
        friend inline const lf operator xor (const point& a, const point& b)
        { return sqrt(pow(a.x - b.x, 2) - pow(a.y - b.y, 2)); }
    };
    static hu n;
    static llu d;
    static point p[N];
    static hu ans;
    inline const lf point::l() const { return x - sqrt(pow(d, 2) - pow(y, 2)); }
    inline const lf point::r() const { return x + sqrt(pow(d, 2) - pow(y, 2)); }
    static inline const void main() {
        scanf("%hu%llu", &n, &d);
        for (hu i(0); i < n; ++i) if (abs(p[i].read().y) > d) return (const void)puts("-1");
        sort(&p[0], &p[n]);
        for (hu i(0); i < n; ++i) {
            if (p[i].vis) continue;
            ++ans;
            for (hu j(i); j < n; ++j) if (p[j].l() <= p[i].r()) p[j].vis = true;
        }
        printf("%hu\n", ans);
    }
}

signed int main() { Main::main(); return 0; }
