[Solution] LOJ #2132 & UOJ#130 & LGP2168 #NOI2015D2A

发布于 2021-07-08 | 分类于 Solution | 5分钟 | 793字数

nnn 个正整数 wiw_iwi​，为每个正整数钦定一字符集为 kkk 的字符串 sis_isi​，使 ∑wi∣si∣\sum w_i|s_i|∑wi​∣si​∣ 最小，求 min⁡∑wi∣si∣\min \sum w_i |s_i|min∑wi​∣si​∣ 与本条件下的 min⁡max⁡∣si∣\min \max |s_i|minmax∣si​∣。

2⩽n⩽1052 \leqslant n \leqslant 10^52⩽n⩽105，2⩽k⩽92 \leqslant k \leqslant 92⩽k⩽9，0<wi⩽10110 < w_i \leqslant 10^{11}0<wi​⩽1011。

考虑一棵 Trie 树对应 sis_isi​，则据题意有所有 sis_isi​ 对应节点皆为叶节点。

设节点 iii 深度为 did_idi​（根节点深度为 000），节点 iii 下方（包括本身）所对应节点的 ∑wi\sum w_i∑wi​ 为 viv_ivi​，

则 ∑wi∣si∣\sum w_i |s_i|∑wi​∣si​∣ 转化为所有非根节点的 ∑vi\sum v_i∑vi​。

考虑在 wiw_iwi​ 结尾补 000，使 n≡1(modk−1)n \equiv 1 \pmod {k - 1}n≡1(modk−1)。

问题转化为：每次合并 kkk 个 wiw_iwi​，代价为被合并的 ∑wi\sum w_i∑wi​，使总代价最小；设 wiw_iwi​ 合并次数为 lil_ili​；求最小总代价，及本条件下的 min⁡max⁡li\min \max l_iminmaxli​。初始 lil_ili​ 为 000。

用堆维护即可。时间复杂度 O((n+k)log⁡n)\mathcal O((n + k) \log n)O((n+k)logn)，空间复杂度 O(n+k)\mathcal O(n + k)O(n+k)。

# include <ciso646>
# include <cstdio>

namespace Main {
    typedef short unsigned int hu;
    typedef unsigned int uint;
    typedef long long unsigned int llu;
    template <typename T>
    static inline const T& max(const T& a, const T& b) { return a < b ? b : a; }
    template <typename T> static inline T& amax(T& a, const T& b) { return a = max(a, b); }
    template <typename T> static inline const void swap(T& a, T& b) { const T c(a); a = b, b = c; }
    template <typename T, const uint N>
    class heap {
        T t[N + 1];
        uint s;
    public:
        heap() : s(0) {}
        inline const void push(const T& x) {
            t[++s] = x; uint u(s);
            while (u > 1 and t[u] < t[u >> 1]) swap(t[u], t[u >> 1]), u >>= 1;
        }
        inline const T top() const { return t[1]; }
        inline const void pop() {
            swap(t[1], t[s--]); uint u(1);
            while (u << 1 <= s && t[u << 1] < t[u] or (u << 1 | 1) <= s && t[u << 1 | 1] < t[u])
                (u << 1 | 1) <= s && t[u << 1 | 1] < t[u << 1] ?
                (swap(t[u], t[u << 1 | 1]), u = u << 1 | 1) : (swap(t[u], t[u << 1]), u <<= 1);
        }
        inline const uint size() const { return s; }
        inline const bool empty() const { return !s; }
    };
    enum { N = (const uint)100000, K = 9 };
    struct sumw {
        llu sum;
        uint len;
        sumw() : sum(0), len(0) {}
        sumw(const llu sum, const uint len = 0) : sum(sum), len(len) {}
        friend inline const bool operator<(const sumw& a, const sumw& b)
        { return a.sum == b.sum ? a.len < b.len : a.sum < b.sum; }
    };
    static uint n;
    static hu k;
    static llu w[N];
    static heap<sumw, N + K> h;
    static llu ans;
    static inline const void main() {
        scanf("%u%hu", &n, &k);
        for (register uint i(0); i < n; ++i) scanf("%llu", &w[i]), h.push(sumw(w[i]));
        for (register uint i(n); (i - 1) % (k - 1); ++i) h.push(sumw(0));
        // printf("s: %u\n", h.size());
        for (register uint i(0); i < (n - 2) / (k - 1) + 1; ++i) {
            llu t(0); uint l(0);
            for (hu j(0); j < k; ++j) t += h.top().sum, amax(l, h.top().len), h.pop();
            ans += t, h.push(sumw(t, l + 1));
            // printf("%llu %u\n", t, l);
            // printf("s: %u\n", h.size());
        }
        printf("%llu\n", ans);
        printf("%u\n", h.top().len);
    }
}

signed int main() { Main::main(); return 0; }
